``````
        - purpose: package manager
    > dotenv
        - version: 16.3.1 (not affected by node.js)

        - version: 1.4.0 (not affected by node.js)
        - purpose: ajax

        - version: 3.0.2 (not affected by node.js)

        - version: 3.10.0 (not affected by node.js)
``````

- start
```
    > local: npm run local
    > dev: npm run dev
    > stage: npm run stage
    > prod: npm run prod
```
